marshaling switch shape, drop the outer `for (size_t r ...)` row loop, write into one
`std::map` not a `std::vector<std::map>`, and pass that single map straight to

```cpp
QUIVER_C_API quiver_error_t quiver_database_update_time_series_group(quiver_database_t* db,
                                                                     const char* collection,
                                                                     const char* group,
                                                                     int64_t id,
                                                                     const char* const* column_names,
                                                                     const int* column_types,
                                                                     const void* const* column_data,
                                                                     size_t column_count,
                                                                     size_t row_count) {
    QUIVER_REQUIRE(db, collection, group);
    if (column_count > 0) {
        QUIVER_REQUIRE(column_names, column_types, column_data);
    }

    try {
        std::vector<std::map<std::string, quiver::Value>> rows;
        rows.reserve(row_count);

        for (size_t r = 0; r < row_count; ++r) {
            std::map<std::string, quiver::Value> row;
            for (size_t c = 0; c < column_count; ++c) {
                std::string col_name(column_names[c]);
                switch (column_types[c]) {
                case QUIVER_DATA_TYPE_INTEGER:
                    row[col_name] = static_cast<const int64_t*>(column_data[c])[r];
                    break;
                case QUIVER_DATA_TYPE_FLOAT:
                    row[col_name] = static_cast<const double*>(column_data[c])[r];
                    break;
                case QUIVER_DATA_TYPE_STRING:
                case QUIVER_DATA_TYPE_DATE_TIME:
                    row[col_name] = std::string(static_cast<const char* const*>(column_data[c])[r]);
                    break;
                default:
                    throw std::runtime_error("Cannot update_time_series_group: unknown column type " +
                                             std::to_string(column_types[c]));
                }
            }
            rows.push_back(std::move(row));
        }

        db->db.update_time_series_group(collection, group, id, rows);
        return QUIVER_OK;
    } catch (const std::exception& e) {
        quiver_set_last_error(e.what());
        return QUIVER_ERROR;
    }
}
```

`column_data[c]` is a pointer to ONE typed value. The cast/dereference shape is:
- INTEGER: `*static_cast<const int64_t*>(column_data[c])`
- FLOAT:   `*static_cast<const double*>(column_data[c])`
- STRING/DATE_TIME: `std::string(*static_cast<const char* const*>(column_data[c]))`

(Equivalent to update's `static_cast<const T*>(column_data[c])[0]`. Either index `[0]` or
deref `*` is acceptable — match update's `[0]` style for visual parity with the sibling
