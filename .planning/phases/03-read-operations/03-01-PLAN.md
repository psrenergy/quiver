    import { Database } from "./database";

    declare module augmentation:
    ```typescript
    declare module "./database" {
      interface Database {
        readScalarIntegers(collection: string, attribute: string): number[];
        readScalarFloats(collection: string, attribute: string): number[];
        readScalarStrings(collection: string, attribute: string): string[];
        readScalarIntegerById(collection: string, attribute: string, id: number): number | null;
        readScalarFloatById(collection: string, attribute: string, id: number): number | null;
        readScalarStringById(collection: string, attribute: string, id: number): string | null;
        readElementIds(collection: string): number[];
      }
    }
    ```

    Implementation details for each method:

    **readScalarIntegers(collection, attribute) -> number[]:**
    - Allocate: `outValues = allocPointerOut()`, `outCount = new BigUint64Array(1)`
    - Call: `check(lib.quiver_database_read_scalar_integers(handle, toCString(collection), toCString(attribute), ptr(outValues), ptr(outCount)))`
    - Read count: `Number(outCount[0])`. If 0, return `[]` immediately (do NOT touch pointer).
    - Read pointer: `readPointerOut(outValues)`
    - Create view: `toArrayBuffer(valuesPtr, 0, count * 8)` then `new BigInt64Array(buffer)`
    - COPY to JS array BEFORE freeing: `result[i] = Number(bigInts[i])` in a loop
    - Free: `lib.quiver_database_free_integer_array(valuesPtr)` -- use raw call (no check()) since free functions return QUIVER_ERROR on null but we already checked count > 0

    **readScalarFloats(collection, attribute) -> number[]:**
    - Same pattern as integers but: `Float64Array` instead of `BigInt64Array`, `free_float_array` instead of `free_integer_array`, `Array.from(new Float64Array(buffer))` for copy.

    **readScalarStrings(collection, attribute) -> string[]:**
    - Same out-param setup. After getting `arrPtr = readPointerOut(outValues)`:
    - Loop: `read.ptr(arrPtr, i * 8)` to get each `char*`, then `new CString(strPtr).toString()` for the string.
    - Free: `lib.quiver_database_free_string_array(arrPtr, count)` -- pass count as the size_t parameter.

    **readScalarIntegerById(collection, attribute, id) -> number | null:**
    - Allocate: `outValue = new BigInt64Array(1)`, `outHasValue = new Int32Array(1)`
    - Call: `check(lib.quiver_database_read_scalar_integer_by_id(handle, toCString(collection), toCString(attribute), id, ptr(outValue), ptr(outHasValue)))`
    - Check: `if (outHasValue[0] === 0) return null`
    - Return: `Number(outValue[0])`

    **readScalarFloatById(collection, attribute, id) -> number | null:**
    - Same pattern but: `outValue = new Float64Array(1)`, read with `outValue[0]` (already a number, no BigInt conversion).

    **readScalarStringById(collection, attribute, id) -> string | null:**
    - Allocate: `outValue = allocPointerOut()` (for char**), `outHasValue = new Int32Array(1)`
    - Call: `check(lib.quiver_database_read_scalar_string_by_id(handle, toCString(collection), toCString(attribute), id, ptr(outValue), ptr(outHasValue)))`
    - Check: `if (outHasValue[0] === 0) return null`
    - Read string: `strPtr = readPointerOut(outValue)`, `result = new CString(strPtr).toString()`
    - Free: `lib.quiver_database_free_string(strPtr)` -- only called when has_value is true (free_string has no null guard)
    - Return: `result`

    **readElementIds(collection) -> number[]:**
    - Same pattern as readScalarIntegers but: only 2 C string args (collection, no attribute), call `quiver_database_read_element_ids`, free with `free_integer_array`.
